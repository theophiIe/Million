int config()
{
	SE_FICHIER fic;
	int *tabGagnants;
	int *tabGain;
	int nbrGagnants = 25;
	
	fic = SE_ouverture ("lottery.cfg", O_RDONLY);

	if (fic.descripteur == -1)
		return -1;
	
	if (SE_lectureEntier(fic, &nbrGagnants) == -1)
		printf("Lecture de nbrGagnants");

	

    tabGagnants = malloc(nbrGagnants * sizeof(int));
    tabGain     = malloc(nbrGagnants * sizeof(int));

    for (int i = 0; i < nbrGagnants; ++i) 
    {
		if (SE_lectureEntier(fic, &tabGagnants[i]) == -1)
			printf("Lecture des numéros gagnants : %d\n", tabGagnants[i]);
    }
		
	return 0;

}
